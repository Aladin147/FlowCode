```typescript
// Define core agentic types
interface AgenticTask {
    id: string;
    goal: string;
    steps: TaskStep[];
    status: TaskStatus;
    approvalRequired: boolean;
    riskLevel: RiskLevel;
}

interface TaskStep {
    id: string;
    action: AgentAction;
    description: string;
    parameters: any;
    dependencies: string[];
    estimatedTime: number;
    riskLevel: RiskLevel;
}

interface AgentAction {
    type: 'analyze' | 'create_file' | 'edit_file' | 'run_command' | 'validate';
    target: string;
    payload: any;
    validation: ValidationRule[];
}
```
